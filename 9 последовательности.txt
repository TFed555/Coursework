@startuml
actor Заведующий as Foo
boundary Окно_редактирования_сотрудника as Foo1
entity База_данных_пользователей as Foo2
control Сообщение_об_ошибке as Foo3 #red
boundary Окно_списка_сотрудников as Foo4

ref over Foo : Нажат элемента из списка

Foo -> Foo1 : Переход из окна списка сотрудников

Foo -> Foo1 : Оставлено/Изменено поле "Отдел", \n нажат чекбокс и кнопка "Подтвердить"
Foo1 -> Foo2 : Изменено поле "Отдел"
Foo2 --> Foo1 : Возврат статуса
Foo1 -> Foo1 : Чекбокс и кнопка "Подтвердить" убираются, \n открывается дополнительное поле
Foo -> Foo1 : Введены данные в поле и нажата кнопка "Подтвердить"

alt Данные корректны
Foo1 -> Foo1 : Добавлено новое поле в профиль
Foo1 -> Foo2 : Сохранить данные
else Некорректные данные
Foo1 -> Foo3 : Ошибка
end

... Альтернативный сценарий ... 
opt
Foo -> Foo1 : Нажата кнопка "Назад"
Foo1 -> Foo4 : Отобразить окно  
end

@enduml
