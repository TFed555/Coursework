@startuml
actor Пользователь as Foo
boundary Окно_авторизации as Foo1
entity База_данных_пользователей as Foo2
control Сообщение_об_ошибке as Foo3 #red
boundary Стартовое_окно_для_роли as Foo4

Foo -> Foo1 : Вводит логин и пароль 
Foo -> Foo1 : Нажимает кнопку Вход
Foo1 -> Foo2 : Проверка на существование логина
Foo2 --> Foo1 : Возврат статуса
alt Логин существует
Foo1 -> Foo2 : Получить роль пользователя
Foo2 --> Foo1 : Возврат роли
Foo1 -> Foo4 : Отобразить стартовое окно (Роль)
else Ошибка ввода данных
Foo1 -> Foo3 : Ошибка ввода данных
end

@enduml
